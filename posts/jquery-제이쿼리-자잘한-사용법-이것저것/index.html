2

1
2

$("#element-id").on("click", function(){...}) // 이벤트 부여
$("#element-id").trigger("click") // 해당 요소를 클릭했을 때의 이벤트 강제 발생

체크박스 체크 상태 변경

1

$("input[type='checkbox'").prop("checked", true) // true: 체크됨, false: 체크 해제

$(".div-wrapper").children()
